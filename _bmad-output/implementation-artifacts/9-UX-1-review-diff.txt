"use client";

import { cn } from "@/lib/utils";
import { type RateGateEvent } from "@/atoms/chat/orchestra.atom";

interface ActivityTimelineProps {
	spawnedAt: number;
	rateGateWaits: RateGateEvent[];
	className?: string;
}

/**
 * Horizontal mini-timeline showing elapsed time and rate-gate pauses.
 * Gate pauses appear as amber ticks so users understand why pacing occurs.
 */
export function ActivityTimeline({ spawnedAt, rateGateWaits, className }: ActivityTimelineProps) {
	const nowMs = Date.now();
	const totalMs = nowMs - spawnedAt;

	if (totalMs < 500) return null;

	return (
		<div
			className={cn("flex flex-col gap-1", className)}
			data-slot="activity-timeline"
			aria-label="Activity timeline"
		>
			{/* Timeline bar */}
			<div className="relative h-1 w-full overflow-hidden rounded-full bg-muted">
				{/* Elapsed fill */}
				<div className="h-full rounded-full bg-muted-foreground/30" style={{ width: "100%" }} />

				{/* Rate-gate ticks */}
				{rateGateWaits.map((gate, i) => {
					const offsetPct = Math.min(((gate.ts - spawnedAt) / totalMs) * 100, 98);
					return (
						<span
							key={i}
							className="absolute top-0 h-full w-0.5 bg-amber-400/80"
							style={{ left: `${offsetPct}%` }}
							title={`Pacing ${gate.waitSeconds}s (${gate.reason})`}
						/>
					);
				})}
			</div>

			{/* Legend — only shown when there are gate waits */}
			{rateGateWaits.length > 0 && (
				<p className="text-[10px] text-muted-foreground/60">
					<span className="inline-block size-1.5 rounded-full bg-amber-400/80 align-middle mr-1" />
					{rateGateWaits.length} pacing pause{rateGateWaits.length > 1 ? "s" : ""}
				</p>
			)}
		</div>
	);
}

"use client";

import { useEffect, useRef, useState } from "react";
import { cn } from "@/lib/utils";

interface LiveNarrationStreamProps {
	text: string | null;
	className?: string;
}

/**
 * Displays the current narration line with a fade-in animation each time
 * the text changes. Keeps only 1 visible line — older text fades out before
 * the new text fades in (cross-fade via key-based remount).
 */
export function LiveNarrationStream({ text, className }: LiveNarrationStreamProps) {
	const [visibleText, setVisibleText] = useState(text);
	const [fadeKey, setFadeKey] = useState(0);
	const prevRef = useRef(text);

	useEffect(() => {
		if (text !== prevRef.current) {
			prevRef.current = text;
			setFadeKey((k) => k + 1);
			setVisibleText(text);
		}
	}, [text]);

	if (!visibleText) return null;

	return (
		<p
			key={fadeKey}
			className={cn("line-clamp-2 animate-fade-in text-muted-foreground", className)}
			aria-live="polite"
			aria-atomic="true"
		>
			{visibleText}
		</p>
	);
}

// ---------------------------------------------------------------------------
// Helpers

type SSEEventInput =
	| { type: string; data?: object; delta?: never }
	| { delta: string; type?: never; data?: never };

function buildSSEStream(events: Array<SSEEventInput>): string {
	return (
		events
			.map((e) => {
				if (e.delta !== undefined) {
					return `data: ${JSON.stringify({ type: "text-delta", delta: e.delta })}\n`;
				}
				return `data: ${JSON.stringify({ type: e.type, data: e.data })}\n`;
			})
			.join("\n") + "\n"
	);
}

const SESSION_ID = "e2e-lab-001";
const AGENT_ID = "tokenomics_analyst";
const AGENT_NAME = "Tokenomics Analyst";
